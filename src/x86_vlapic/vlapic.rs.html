120    /// IF any bits set in VISR
121    ///     THEN SVI := highest index of bit set in VISR
122    ///     ELSE SVI := 0;
123    /// FI;
124    fn find_isrv(&self) -> u32 {
125        let mut isrv = 0;
126        /* i ranges effectively from 7 to 1 */
127        for i in (1..8).rev() {
128            let val = self.regs().ISR[i].get() as u32;
129            if val != 0 {
130                isrv = ((i as u32) << 5) | fls32(val) as u32;
131                break;
132            }
133        }
134
135        isrv
136    }

138    fn update_ppr(&mut self) {
139        let isrv = self.isrv;
140        let tpr = self.regs().TPR.get() as u32;
141        // IF VTPR[7:4] ≥ SVI[7:4]
142        let ppr = if prio(tpr) >= prio(isrv) {
143            // THEN VPPR := VTPR & FFH;
144            tpr
145        } else {
146            // ELSE VPPR := SVI & F0H;
147            isrv & 0xf0
148        };
149        self.regs().PPR.set(ppr as _);
150    }

152    /// Process the EOI operation triggered by a write to the EOI register.
153    /// 11.8.5 Signaling Interrupt Servicing Completion
154    /// 30.1.4 EOI Virtualization
155    fn process_eoi(&mut self) {
156        let vector = self.isrv;
157
158        if vector == 0 {
159            return;
160        }
161
162        let (idx, bitpos) = extract_index_and_bitpos_u32(vector);
163
164        // Upon receiving an EOI, the APIC clears the highest priority bit in the ISR
165        // and dispatches the next highest priority interrupt to the processor.
166
167        // VISR[Vector] := 0; (see Section 30.1.1 for definition of VISR)
168        let mut isr = self.regs().ISR[idx].get();
169        isr &= !(1 << bitpos);
170        self.regs().ISR[idx].set(isr);
171
172        // IF any bits set in VISR
173        // THEN SVI := highest index of bit set in VISR
174        // ELSE SVI := 0;
175        self.isrv = self.find_isrv();
176
177        // perform PPR virtualiation (see Section 30.1.3);
178        self.update_ppr();
179
180        // The trigger mode register (TMR) indicates the trigger mode of the interrupt (see Figure 11-20).
181        // Upon acceptance of an interrupt into the IRR, the corresponding TMR bit is cleared for edge-triggered interrupts and set for leveltriggered interrupts.
182        // If a TMR bit is set when an EOI cycle for its corresponding interrupt vector is generated, an EOI message is sent to all I/O APICs.
183        // (see 11.8.4 Interrupt Acceptance for Fixed Interrupts)
184        if (self.regs().TMR[idx].get() as u32).bit(bitpos) {
185            // Send EOI to all I/O APICs
186            /*
187             * Per Intel SDM 10.8.5, Software can inhibit the broadcast of
188             * EOI by setting bit 12 of the Spurious Interrupt Vector
189             * Register of the LAPIC.
190             * TODO: Check if the bit 12 "Suppress EOI Broadcasts" is set.
191             */
192            unimplemented!("vioapic_broadcast_eoi(vlapic2vcpu(vlapic)->vm, vector);")
193        }
194
195        debug!("Gratuitous EOI vector: {:#010X}", vector);
196
197        unimplemented!("vcpu_make_request(vlapic2vcpu(vlapic), ACRN_REQUEST_EVENT);")
198    }

221    fn is_dest_field_matched(&self, dest: u32) -> AxResult<bool> {
222        let mut ret = false;
223
224        let ldr = self.regs().LDR.get();
225
226        if self.is_x2apic_enabled() {
227            return Ok(true);
228        } else {
229            match self
230                .regs()
231                .DFR
232                .read_as_enum::<APICDestinationFormat>(DESTINATION_FORMAT::Model)
233                .ok_or(AxError::InvalidData)?
234            {
235                APICDestinationFormat::Flat => {
236                    /*
237                     * In the "Flat Model" the MDA is interpreted as an 8-bit wide
238                     * bitmask. This model is available in the xAPIC mode only.
239                     */
240                    let logical_id = ldr >> 24;
241                    let dest_logical_id = dest & 0xff;
242                    if logical_id & dest_logical_id != 0 {
243                        ret = true;
244                    }
245                }
246                APICDestinationFormat::Cluster => {
247                    /*
248                     * In the "Cluster Model" the MDA is used to identify a
249                     * specific cluster and a set of APICs in that cluster.
250                     */
251                    let logical_id = (ldr >> 24) & 0xf;
252                    let cluster_id = ldr >> 28;
253                    let dest_logical_id = dest & 0xf;
254                    let dest_cluster_id = (dest >> 4) & 0xf;
255                    if (cluster_id == dest_cluster_id) && ((logical_id & dest_logical_id) != 0) {
256                        ret = true;
257                    }
258                }
259            }
260        }
261        Ok(ret)
262    }

264    /// This function populates 'dmask' with the set of vcpus that match the
265    /// addressing specified by the (dest, phys, lowprio) tuple.
266    fn calculate_dest_no_shorthand(
267        &self,
268        is_broadcast: bool,
269        dest: u32,
270        is_phys: bool,
271        lowprio: bool,
272    ) -> AxResult<u64> {
273        let mut dmask = 0;
274
275        if is_broadcast {
276            // Broadcast in both logical and physical modes.
277            dmask = vmm::current_vm_active_vcpus() as u64;
278        } else if is_phys {
279            // Physical mode: "dest" is local APIC ID.
280            // Todo: distinguish between APIC ID and vCPU ID.
281            dmask = 1 << dest;
282        } else if lowprio {
283            // lowprio is not supported.
284            // Refer to 11.6.2.4 Lowest Priority Delivery Mode.
285            unimplemented!("lowprio");
286        } else {
287            // Logical mode: "dest" is message destination addr
288            // to be compared with the logical APIC ID in LDR.
289
290            let vcpu_mask = vmm::active_vcpus(vmm::current_vm_id()).unwrap();
291            for i in 0..vmm::current_vm_vcpu_num() {
292                if vcpu_mask & (1 << i) != 0 {
293                    if !self.is_dest_field_matched(dest)? {
294                        continue;
295                    }
296                    dmask |= 1 << i;
297                }
298            }
299        }
300
301        Ok(dmask)
302    }

304    fn calculate_dest(
305        &self,
306        shorthand: APICDestination,
307        is_broadcast: bool,
308        dest: u32,
309        is_phys: bool,
310        lowprio: bool,
311    ) -> AxResult<u64> {
312        let mut dmask = 0;
313        match shorthand {
314            APICDestination::NoShorthand => {
315                dmask = self.calculate_dest_no_shorthand(is_broadcast, dest, is_phys, lowprio)?;
316            }
317            APICDestination::SELF => {
318                dmask.set_bit(self.vapic_id as usize, true);
319            }
320            APICDestination::AllIncludingSelf => {
321                dmask = vmm::current_vm_active_vcpus() as u64;
322            }
323            APICDestination::AllExcludingSelf => {
324                dmask = vmm::current_vm_active_vcpus() as u64;
325                dmask &= !(1 << self.vapic_id);
326            }
327        }
328
329        Ok(dmask)
330    }

378    fn write_dfr(&mut self) {
379        use crate::regs::DESTINATION_FORMAT;
380
381        const APIC_DFR_RESERVED: u32 = 0x0fff_ffff;
382        const APIC_DFR_MODEL_MASK: u32 = 0xf000_0000;
383
384        let mut dfr = self.regs().DFR.get();
385        dfr &= APIC_DFR_MODEL_MASK;
386        dfr |= APIC_DFR_RESERVED;
387        self.regs().DFR.set(dfr);
388
389        debug!("[VLAPIC] write DFR register to {:#010X}", dfr);
390
391        match self.regs().DFR.read_as_enum(DESTINATION_FORMAT::Model) {
392            Some(DESTINATION_FORMAT::Model::Value::Flat) => {
393                debug!("[VLAPIC] DFR in Flat Model");
394            }
395            Some(DESTINATION_FORMAT::Model::Value::Cluster) => {
396                debug!("[VLAPIC] DFR in Cluster Model");
397            }
398            None => {
399                debug!("[VLAPIC] DFR in Unknown Model {:#010X}", dfr);
400            }
401        }

404    /// Figure 11-14. Spurious-Interrupt Vector Register (SVR)
405    /// Handle writes to the SVR register.
406    fn write_svr(&mut self) -> AxResult {
407        let new = self.regs().SVR.extract();
408        let old = self.svr_last;
409
410        self.svr_last = new;
411
412        if old.is_set(SPURIOUS_INTERRUPT_VECTOR::APICSoftwareEnableDisable)
413            && !new.is_set(SPURIOUS_INTERRUPT_VECTOR::APICSoftwareEnableDisable)
414        {
415            debug!("[VLAPIC] vlapic [{}] is software-disabled", self.vapic_id);
416            // The apic is now disabled so stop the apic timer
417            // and mask all the LVT entries.
418            self.virtual_timer.stop_timer()?;
419            self.mask_lvts()?;
420            warn!("VM wire mode should be changed to INTR here, unimplemented");
421        } else if !old.is_set(SPURIOUS_INTERRUPT_VECTOR::APICSoftwareEnableDisable)
422            && new.is_set(SPURIOUS_INTERRUPT_VECTOR::APICSoftwareEnableDisable)
423        {
424            debug!("[VLAPIC] vlapic [{}] is software-enabled", self.vapic_id);
425
426            // The apic is now enabled so restart the apic timer
427            // if it is configured in periodic mode.
428            if self.virtual_timer.is_periodic() {
429                debug!("Restarting the apic timer");
430                self.virtual_timer.restart_timer()?;
431            }
432        }
433
434        Ok(())
435    }

444    fn write_icr(&mut self) -> AxResult {
445        self.regs()
446            .ICR_LO
447            .modify(INTERRUPT_COMMAND_LOW::DeliveryStatus::Idle);
448
449        let icr_low = self.regs().ICR_LO.extract();
450
451        let (dest, is_broadcast) = if self.is_x2apic_enabled() {
452            use crate::consts::x2apic::X2APIC_BROADCAST_DEST_ID;
453            let dest = self.regs().ICR_HI.get();
454            (dest, dest == X2APIC_BROADCAST_DEST_ID)
455        } else {
456            use crate::consts::xapic::XAPIC_BROADCAST_DEST_ID;
457            let dest = self.regs().ICR_HI.read(INTERRUPT_COMMAND_HIGH::Destination);
458            (dest, dest == XAPIC_BROADCAST_DEST_ID)
459        };
460
461        let vec = icr_low.read(INTERRUPT_COMMAND_LOW::Vector);
462        let mode = icr_low
463            .read_as_enum::<APICDeliveryMode>(INTERRUPT_COMMAND_LOW::DeliveryMode)
464            .ok_or(AxError::InvalidData)?;
465        let is_phys = icr_low.is_set(INTERRUPT_COMMAND_LOW::DestinationMode);
466        let shorthand = icr_low
467            .read_as_enum::<APICDestination>(INTERRUPT_COMMAND_LOW::DestinationShorthand)
468            .ok_or(AxError::InvalidData)?;
469
470        if mode == APICDeliveryMode::Fixed && vec < 16 {
471            self.set_err(ERROR_STATUS::SendIllegalVector::SET);
472            debug!("[VLAPIC] Ignoring invalid IPI {:#010X}", vec);
473        } else if (shorthand == APICDestination::SELF
474            || shorthand == APICDestination::AllIncludingSelf)
475            && (mode == APICDeliveryMode::NMI
476                || mode == APICDeliveryMode::INIT
477                || mode == APICDeliveryMode::StartUp)
478        {
479            debug!("[VLAPIC] Invalid ICR value {:#010X}", vec);
480        } else {
481            debug!(
482                "icrlow {:#010X} icrhi {:#010X} triggered ipi {:#010X}",
483                icr_low.get(),
484                self.regs().ICR_HI.get(),
485                vec
486            );
487            let dmask = self.calculate_dest(shorthand, is_broadcast, dest, is_phys, false)?;
488
489            // TODO: we need to get the specific vcpu number somehow.
490            for i in 0..vmm::current_vm_vcpu_num() as u32 {
491                if dmask & (1 << i) != 0 {
492                    match mode {
493                        APICDeliveryMode::Fixed => {
494                            self.set_intr(i, vec, LAPIC_TRIG_EDGE);
495                            debug!("[VLAPIC] sending IPI {} to vcpu {}", vec, i);
496                        }
497                        APICDeliveryMode::NMI => {
498                            self.inject_nmi(i);
499                            debug!("[VLAPIC] sending NMI to vcpu {}", i);
500                        }
501                        APICDeliveryMode::INIT | APICDeliveryMode::StartUp => {
502                            self.process_init_sipi(i, mode, icr_low);
503                        }
504                        APICDeliveryMode::SMI => {
505                            warn!("[VLPAIC] SMI IPI do not support");
506                        }
507                        _ => {
508                            error!("Unhandled icrlo write with mode {:?}\n", mode);
509                        }
510                    }
511                }
512            }
513        }
514
515        Ok(())
516    }

534    fn write_lvt(&mut self, offset: ApicRegOffset) -> AxResult {
535        let mut val = self.extract_lvt_val(offset);
536
537        if self
538            .regs()
539            .SVR
540            .is_set(SPURIOUS_INTERRUPT_VECTOR::APICSoftwareEnableDisable)
541        {
542            val |= APIC_LVT_M;
543        }
544
545        // Mask::Masked, Delivery Status:SendPending, Vector::SET(0xff)
546        let mut mask = APIC_LVT_M | APIC_LVT_DS | APIC_LVT_VECTOR;
547
548        match offset {
549            ApicRegOffset::LvtTimer => {
550                mask |= LVT_TIMER::TimerMode::SET.mask();
551                val &= mask;
552                self.regs().LVT_TIMER.set(val); // Duplicated, which one should be removed?
553                self.lvt_last.lvt_timer.set(val);
554
555                self.virtual_timer.write_lvt(val)?;
556            }
557            ApicRegOffset::LvtErr => {
558                val &= mask;
559                self.regs().LVT_ERROR.set(val);
560                self.lvt_last.lvt_err.set(val);
561            }
562            ApicRegOffset::LvtLint0 => {
563                mask |= LVT_LINT0::TriggerMode::SET.mask();
564                mask |= LVT_LINT0::RemoteIRR::SET.mask();
565                mask |= LVT_LINT0::InterruptInputPinPolarity::SET.mask();
566                mask |= LVT_LINT0::DeliveryMode::SET.mask();
567                val &= mask;
568
569                // vlapic mask/unmask LINT0 for ExtINT?
570                if (val & LVT_LINT0::DeliveryMode::SET.mask())
571                    == LVT_LINT0::DeliveryMode::ExtINT.mask()
572                {
573                    let last = self.lvt_last.lvt_lint0;
574                    if last.is_set(LVT_LINT0::Mask) && val & LVT_LINT0::Mask::SET.mask() == 0 {
575                        // mask -> unmask: may from every vlapic in the vm
576                        warn!("vpic wire mode change to LAPIC, unimplemented");
577                    } else if !last.is_set(LVT_LINT0::Mask)
578                        && val & LVT_LINT0::Mask::SET.mask() != 0
579                    {
580                        // unmask -> mask: only from the vlapic LINT0-ExtINT enabled
581                        warn!("vpic wire mode change to NULL, unimplemented");
582                    } else {
583                        // APIC_LVT_M unchanged. No action required.
584                    }
585                }
586
587                self.regs().LVT_LINT0.set(val);
588                self.lvt_last.lvt_lint0.set(val);
589            }
590            ApicRegOffset::LvtLint1 => {
591                mask |= LVT_LINT1::TriggerMode::SET.mask();
592                mask |= LVT_LINT1::RemoteIRR::SET.mask();
593                mask |= LVT_LINT1::InterruptInputPinPolarity::SET.mask();
594                mask |= LVT_LINT1::DeliveryMode::SET.mask();
595                val &= mask;
596
597                self.regs().LVT_LINT1.set(val);
598                self.lvt_last.lvt_lint1.set(val);
599            }
600            ApicRegOffset::LvtCMCI => {
601                mask |= LVT_CMCI::DeliveryMode::SET.mask();
602                val &= mask;
603                self.regs().LVT_CMCI.set(val);
604                self.lvt_last.lvt_cmci.set(val);
605            }
606            ApicRegOffset::LvtPmc => {
607                mask |= LVT_PERFORMANCE_COUNTER::DeliveryMode::SET.mask();
608                val &= mask;
609                self.regs().LVT_PMI.set(val);
610                self.lvt_last.lvt_perf_count.set(val);
611            }
612            ApicRegOffset::LvtThermal => {
613                mask |= LVT_THERMAL_MONITOR::DeliveryMode::SET.mask();
614                val &= mask;
615                self.regs().LVT_THERMAL.set(val);
616                self.lvt_last.lvt_thermal.set(val);
617            }
618            _ => {
619                warn!("[VLAPIC] write unsupported APIC register: {:?}", offset);
620                return Err(AxError::InvalidInput);
621            }
622        }
623        Ok(())
624    }
625
626    fn mask_lvts(&mut self) -> AxResult {
627        self.regs().LVT_CMCI.modify(LVT_CMCI::Mask::SET);
628        self.write_lvt(ApicRegOffset::LvtCMCI)?;
629
630        self.regs().LVT_TIMER.modify(LVT_TIMER::Mask::SET);
631        self.write_lvt(ApicRegOffset::LvtTimer)?;
632
633        self.regs()
634            .LVT_THERMAL
635            .modify(LVT_THERMAL_MONITOR::Mask::SET);
636        self.write_lvt(ApicRegOffset::LvtThermal)?;
637
638        self.regs()
639            .LVT_PMI
640            .modify(LVT_PERFORMANCE_COUNTER::Mask::SET);
641        self.write_lvt(ApicRegOffset::LvtPmc)?;
642
643        self.regs().LVT_LINT0.modify(LVT_LINT0::Mask::SET);
644        self.write_lvt(ApicRegOffset::LvtLint0)?;
645
646        self.regs().LVT_LINT1.modify(LVT_LINT1::Mask::SET);
647        self.write_lvt(ApicRegOffset::LvtLint1)?;
648
649        self.regs().LVT_ERROR.modify(LVT_ERROR::Mask::SET);
650        self.write_lvt(ApicRegOffset::LvtErr)?;
651
652        Ok(())
653    }

655    fn write_icrtmr(&mut self) -> AxResult {
656        self.virtual_timer.write_icr(self.regs().ICR_TIMER.get())
657    }

659    fn write_dcr(&mut self) -> AxResult {
660        self.virtual_timer.write_dcr(self.regs().DCR_TIMER.get());
661        Ok(())
662    }

680impl VirtualApicRegs {
681    pub fn handle_read(&self, offset: ApicRegOffset, width: AccessWidth) -> AxResult<usize> {
682        let mut value: usize = 0;
683        match offset {
684            ApicRegOffset::ID => {
685                value = self.regs().ID.get() as _;
686            }
687            ApicRegOffset::Version => {
688                value = self.regs().VERSION.get() as _;
689            }
690            ApicRegOffset::TPR => {
691                value = self.regs().TPR.get() as _;
692            }
693            ApicRegOffset::PPR => {
694                value = self.regs().PPR.get() as _;
695            }
696            ApicRegOffset::EOI => {
697                // value = self.regs().EOI.get() as _;
698                warn!("[VLAPIC] read EOI register: {:#010X}", value);
699            }
700            ApicRegOffset::LDR => {
701                value = self.regs().LDR.get() as _;
702            }
703            ApicRegOffset::DFR => {
704                value = self.regs().DFR.get() as _;
705            }
706            ApicRegOffset::SIVR => {
707                value = self.regs().SVR.get() as _;
708            }
709            ApicRegOffset::ISR(index) => {
710                value = self.regs().ISR[index.as_usize()].get() as _;
711            }
712            ApicRegOffset::TMR(index) => {
713                value = self.regs().TMR[index.as_usize()].get() as _;
714            }
715            ApicRegOffset::IRR(index) => {
716                value = self.regs().IRR[index.as_usize()].get() as _;
717            }
718            ApicRegOffset::ESR => {
719                value = self.regs().ESR.get() as _;
720            }
721            ApicRegOffset::ICRLow => {
722                value = self.regs().ICR_LO.get() as _;
723                if self.is_x2apic_enabled() && width == AccessWidth::Qword {
724                    let icr_hi = self.regs().ICR_HI.get() as usize;
725                    value |= icr_hi << 32;
726                    debug!("[VLAPIC] read ICR register: {:#018X}", value);
727                } else if self.is_x2apic_enabled() ^ (width == AccessWidth::Qword) {
728                    warn!(
729                        "[VLAPIC] Illegal read attempt of ICR register at width {:?} with X2APIC {}",
730                        width,
731                        if self.is_x2apic_enabled() {
732                            "enabled"
733                        } else {
734                            "disabled"
735                        }
736                    );
737                    return Err(AxError::InvalidInput);
738                }
739            }
740            ApicRegOffset::ICRHi => {
741                value = self.regs().ICR_HI.get() as _;
742            }
743            // Local Vector Table registers.
744            ApicRegOffset::LvtCMCI => {
745                value = self.lvt_last.lvt_cmci.get() as _;
746            }
747            ApicRegOffset::LvtTimer => {
748                value = self.lvt_last.lvt_timer.get() as _;
749            }
750            ApicRegOffset::LvtThermal => {
751                value = self.lvt_last.lvt_thermal.get() as _;
752            }
753            ApicRegOffset::LvtPmc => {
754                value = self.lvt_last.lvt_perf_count.get() as _;
755            }
756            ApicRegOffset::LvtLint0 => {
757                value = self.lvt_last.lvt_lint0.get() as _;
758            }
759            ApicRegOffset::LvtLint1 => {
760                value = self.lvt_last.lvt_lint1.get() as _;
761            }
762            ApicRegOffset::LvtErr => {
763                value = self.lvt_last.lvt_err.get() as _;
764            }
765            // Timer registers.
766            ApicRegOffset::TimerInitCount => {
767                match self.timer_mode() {
768                    Ok(TimerMode::OneShot) | Ok(TimerMode::Periodic) => {
769                        value = self.regs().ICR_TIMER.get() as _;
770                    }
771                    Ok(TimerMode::TSCDeadline) => {
772                        /* if TSCDEADLINE mode always return 0*/
773                        value = 0;
774                    }
775                    _ => {
776                        warn!("[VLAPIC] read TimerInitCount register: invalid timer mode");
777                    }
778                }
779                debug!("[VLAPIC] read TimerInitCount register: {:#010X}", value);
780            }
781            ApicRegOffset::TimerCurCount => {
782                value = self.virtual_timer.read_ccr() as _;
783            }
784            ApicRegOffset::TimerDivConf => {
785                value = self.regs().DCR_TIMER.get() as _;
786            }
787            _ => {
788                warn!("[VLAPIC] read unknown APIC register: {:?}", offset);
789            }
790        }
791        debug!("[VLAPIC] read {} register: {:#010X}", offset, value);
792        Ok(value)
793    }
794
795    pub fn handle_write(
796        &mut self,
797        offset: ApicRegOffset,
798        val: usize,
799        width: AccessWidth,
800    ) -> AxResult {
801        let data32 = val as u32;
802
803        match offset {
804            ApicRegOffset::ID => {
805                // Force APIC ID to be read-only.
806                // self.regs().ID.set(val as _);
807            }
808            ApicRegOffset::EOI => {
809                self.process_eoi();
810            }
811            ApicRegOffset::LDR => {
812                self.regs().LDR.set(data32);
813                self.write_ldr();
814            }
815            ApicRegOffset::DFR => {
816                self.regs().DFR.set(data32);
817                self.write_dfr();
818            }
819            ApicRegOffset::SIVR => {
820                self.regs().SVR.set(data32);
821                self.write_svr()?;
822            }
823            ApicRegOffset::ESR => {
824                self.regs().ESR.set(data32);
825                self.write_esr();
826            }
827            ApicRegOffset::ICRLow => {
828                if self.is_x2apic_enabled() && width == AccessWidth::Qword {
829                    debug!("[VLAPIC] write ICR register: {:#018X} in X2APIC mode", val);
830                    self.regs().ICR_HI.set((val >> 32) as u32);
831                } else if self.is_x2apic_enabled() ^ (width == AccessWidth::Qword) {
832                    warn!(
833                        "[VLAPIC] Illegal read attempt of ICR register at width {:?} with X2APIC {}",
834                        width,
835                        if self.is_x2apic_enabled() {
836                            "enabled"
837                        } else {
838                            "disabled"
839                        }
840                    );
841                    return Err(AxError::InvalidInput);
842                }
843                self.regs().ICR_LO.set(data32);
844                self.write_icr()?;
845            }
846            // Local Vector Table registers.
847            ApicRegOffset::LvtCMCI => {
848                self.regs().LVT_CMCI.set(data32);
849                self.write_lvt(offset)?;
850            }
851            ApicRegOffset::LvtTimer => {
852                self.regs().LVT_TIMER.set(data32);
853                self.write_lvt(offset)?;
854            }
855            ApicRegOffset::LvtThermal => {
856                self.regs().LVT_THERMAL.set(data32);
857                self.write_lvt(offset)?;
858            }
859            ApicRegOffset::LvtPmc => {
860                self.regs().LVT_PMI.set(data32);
861                self.write_lvt(offset)?;
862            }
863            ApicRegOffset::LvtLint0 => {
864                self.regs().LVT_LINT0.set(data32);
865                self.write_lvt(offset)?;
866            }
867            ApicRegOffset::LvtLint1 => {
868                self.regs().LVT_LINT1.set(data32);
869                self.write_lvt(offset)?;
870            }
871            ApicRegOffset::LvtErr => {
872                self.regs().LVT_ERROR.set(data32);
873                self.write_lvt(offset)?;
874            }
875            // Timer registers.
876            ApicRegOffset::TimerInitCount => {
877                // if TSCDEADLINE mode ignore icr_timer
878                if self.timer_mode()? == TimerMode::TSCDeadline {
879                    warn!(
880                        "[VLAPIC] write TimerInitCount register: ignore icr_timer in TSCDEADLINE mode"
881                    );
882                    return Ok(());
883                }
884                self.regs().ICR_TIMER.set(data32);
885                self.write_icrtmr()?;
886            }
887            ApicRegOffset::TimerDivConf => {
888                self.regs().DCR_TIMER.set(data32);
889                self.write_dcr()?;
890            }
891            ApicRegOffset::SelfIPI => {
892                if self.is_x2apic_enabled() {
893                    self.regs().SELF_IPI.set(data32);
894                    self.handle_self_ipi();
895                } else {
896                    warn!("[VLAPIC] write SelfIPI register: unsupported in xAPIC mode");
897                    return Err(AxError::InvalidInput);
898                }
899            }
900            _ => {
901                warn!("[VLAPIC] write unsupported APIC register: {:?}", offset);
902                return Err(AxError::InvalidInput);
903            }
904        }
905
906        debug!("[VLAPIC] write {} register: {:#010X}", offset, val);
907
908        Ok(())
909    }
