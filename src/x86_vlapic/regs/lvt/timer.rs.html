1use tock_registers::LocalRegisterCopy;
2use tock_registers::register_bitfields;
3use tock_registers::registers::ReadWrite;

5register_bitfields! {
6    u32,
7    pub LVT_TIMER [
8        /// Reserved2
9        Reserved2 OFFSET(19) NUMBITS(13) [],
10        /// Timer Mode
11        TimerMode OFFSET(17) NUMBITS(2) [
12            /// (00b) one-shot mode using a count-down value
13            OneShot = 0b00,
14            /// (01b) periodic mode reloading a count-down value
15            Periodic = 0b01,
16            /// (10b) TSC-Deadline mode using absolute target value in IA32_TSC_DEADLINE MSR (see Section 11.5.4.1)
17            TSCDeadline = 0b10,
18            /// (11b) is reserved
19            Reserved = 0b11
20        ],
21        /// Mask: Interrupt mask:
22        /// (0) enables reception of the interrupt and (1) inhibits reception of the interrupt.
23        /// When the local APIC handles a performance-monitoring counters interrupt,
24        /// it automatically sets the mask flag in the LVT performance counter register.
25        /// This flag is set to 1 on reset.
26        /// It can be cleared only by software.
27        Mask OFFSET(16) NUMBITS(1) [
28            /// Not masked, enables reception of the interrupt.
29            NotMasked = 0,
30            /// Masked, inhibits reception of the interrupt.
31            Masked = 1
32        ],
33        Reserved1 OFFSET(13) NUMBITS(3) [],
34        /// Delivery Status (Read Only): Indicates the interrupt delivery status
35        DeliveryStatus OFFSET(12) NUMBITS(1) [
36            /// 0 (Idle)
37            /// There is currently no activity for this interrupt source,
38            /// or the previous interrupt from this source was delivered to the processor core and accepted.
39            Idle = 0,
40            /// 1 (Send Pending)
41            /// Indicates that an interrupt from this source has been delivered to the processor core
42            /// but has not yet been accepted (see Section 11.5.5, “Local Interrupt Acceptance”).
43            SendPending = 1
44        ],
45        Reserved0 OFFSET(8) NUMBITS(4) [],
46        /// Vector: Interrupt vector number.
47        Vector OFFSET(0) NUMBITS(8) [],
48    ]
49}
