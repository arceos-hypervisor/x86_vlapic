1//! 11.6.1 Interrupt Command Register (ICR)
2//! The interrupt command register (ICR) is a 64-bit1 local APIC register (see Figure 11-12)
3//! that allows software running on the processor to specify and send interprocessor interrupts (IPIs) to other processors in the system.
4
5use tock_registers::LocalRegisterCopy;
6use tock_registers::register_bitfields;
7use tock_registers::registers::ReadWrite;
8
9register_bitfields! {
10    u32,
11    pub INTERRUPT_COMMAND_LOW [
12        /// Reserved
13        Reserved2 OFFSET(20) NUMBITS(12) [],
14        /// Destination Shorthand
15        /// Indicates whether a shorthand notation is used to specify the destination of the interrupt and, if so, which shorthand is used. Destination shorthands are used in place of the 8-bit destination field, and can be sent by software using a single write to the low doubleword of the ICR. Shorthands are defined for the following cases: software self interrupt, IPIs to all processors in the system including the sender, IPIs to all processors in the system excluding the sender.
16        /// - 00: (No Shorthand)
17        ///     The destination is specified in the destination field.
18        /// - 01: (Self)
19        ///     The issuing APIC is the one and only destination of the IPI. This destination shorthand allows software to interrupt the processor on which it is executing. An APIC implementation is free to deliver the self-interrupt message internally or to issue the message to the bus and “snoop” it as with any other IPI message.
20        /// - 10: (All Including Self)
21        ///     The IPI is sent to all processors in the system including the processor sending the IPI. The APIC will broadcast an IPI message with the destination field set to FH for Pentium and P6 family processors and to FFH for Pentium 4 and Intel Xeon processors.
22        /// - 11: (All Excluding Self)
23        ///     The IPI is sent to all processors in a system with the exception of the processor sending the IPI. The APIC broadcasts a message with the physical destination mode and destination field set to FH for Pentium and P6 family processors and to FFH for Pentium 4 and Intel Xeon processors. Support for this destination shorthand in conjunction with the lowest-priority delivery mode is model specific. For Pentium 4 and Intel Xeon processors, when this shorthand is used together with lowest priority delivery mode, the IPI may be redirected back to the issuing processor.
24        DestinationShorthand OFFSET(18) NUMBITS(2) [
25            /// No Shorthand
26            NoShorthand = 0b00,
27            /// Self
28            SELF = 0b01,
29            /// All Including Self
30            AllIncludingSelf = 0b10,
31            /// All Excluding Self
32            AllExcludingSelf = 0b11
33        ],
34        /// Reserved
35        Reserved1 OFFSET(16) NUMBITS(2) [],
36        /// Trigger Mode
37        /// Selects the trigger mode when using the INIT level de-assert delivery mode:
38        ///     edge (0) or level (1).
39        /// It is ignored for all other delivery modes.
40        /// (This flag has no meaning in Pentium 4 and Intel Xeon processors, and will always be issued as a 0.)
41        TriggerMode OFFSET(15) NUMBITS(1) [
42            /// Edge
43            Edge = 0,
44            /// Level
45            Level = 1
46        ],
47        /// Level
48        /// For the INIT level de-assert delivery mode this flag must be set to 0;
49        /// for all other delivery modes it must be set to 1.
50        ///  (This flag has no meaning in Pentium 4 and Intel Xeon processors, and will always be issued as a 1.)
51        Level OFFSET(14) NUMBITS(1) [
52            /// De-assert
53            DeAssert = 0,
54            /// Assert
55            Assert = 1
56        ],
57        /// Reserved
58        Reserved0 OFFSET(13) NUMBITS(1) [],
59        /// Delivery Status (Read Only) Indicates the IPI delivery status, as follows:
60        /// - 0 (Idle) Indicates that this local APIC has completed sending any previous IPIs.
61        /// - 1 (Send Pending) Indicates that this local APIC has not completed sending the last IPI.
62        DeliveryStatus OFFSET(12) NUMBITS(1) [
63            /// Idle
64            Idle = 0,
65            /// Send Pending
66            SendPending = 1
67        ],
68        /// Destination Mode Selects either physical (0) or logical (1) destination mode (see Section 11.6.2, “Determining IPI Destination”).
69        DestinationMode OFFSET(11) NUMBITS(1) [
70            /// Physical
71            Physical = 0,
72            /// Logical
73            Logical = 1
74        ],
75        /// Delivery Mode Specifies the type of IPI to be sent.
76        /// This field is also know as the IPI message type field.
77        /// - 000 (Fixed)
78        ///     Delivers the interrupt specified in the vector field to the target processor or processors.
79        /// - 001 (Lowest Priority)
80        ///     Same as fixed mode, except that the interrupt is delivered to the processor executing at the lowest priority among the set of processors specified in the destination field. The ability for a processor to send a lowest priority IPI is model specific and should be avoided by BIOS and operating system software.
81        /// - 010 (SMI)
82        ///     Delivers an SMI interrupt to the target processor or processors. The vector field must be programmed to 00H for future compatibility.
83        /// - 011 (Reserved)
84        /// - 100 (NMI)
85        ///     Delivers an NMI interrupt to the target processor or processors. The vector information is ignored.
86        /// - 101 (INIT)
87        ///     Delivers an INIT request to the target processor or processors, which causes them to perform an INIT.
88        ///     As a result of this IPI message, all the target processors perform an INIT.
89        ///     The vector field must be programmed to 00H for future compatibility.
90        /// - 101 (INIT Level De-assert)
91        ///     (Not supported in the Pentium 4 and Intel Xeon processors.)
92        ///     Sends a synchronization message to all the local APICs in the system to set their arbitration IDs (stored in their Arb ID registers) to the values of their APIC IDs (see Section 11.7, “System and APIC Bus Arbitration”).
93        ///     For this delivery mode, the level flag must be set to 0 and trigger mode flag to 1.
94        ///     This IPI is sent to all processors, regardless of the value in the destination field or the destination shorthand field;
95        ///     however, software should specify the “all including self” shorthand.
96        /// - 110 (Start-Up)
97        ///     Sends a special “start-up” IPI (called a SIPI) to the target processor or processors.
98        ///     The vector typically points to a start-up routine that is part of the BIOS boot-strap code
99        ///     (see Section 9.4, “Multiple-Processor (MP) Initialization”).
100        ///     IPIs sent with this delivery mode are not automatically retried if the source APIC is unable to deliver it.
101        ///     It is up to the software to determine if the SIPI was not successfully delivered and to reissue the SIPI if necessary.
102        /// - 111 (Reserved)
103        DeliveryMode OFFSET(8) NUMBITS(3) [
104            /// Fixed
105            Fixed = 0b000,
106            /// Lowest Priority
107            LowestPriority = 0b001,
108            /// SMI
109            SMI = 0b010,
110            /// Reserved
111            Reserved011 = 0b011,
112            /// NMI
113            NMI = 0b100,
114            /// INIT
115            INIT = 0b101,
116            /// Start-Up
117            StartUp = 0b110,
118            /// Start-Up
119            Reserved111 = 0b111
120        ],
121        /// Vector The vector number of the interrupt being sent.
122        Vector OFFSET(0) NUMBITS(8) []
123    ]
124}

126register_bitfields! {
127    u32,
128    pub INTERRUPT_COMMAND_HIGH [
129        /// Destination [56:63]
130        /// Specifies the target processor or processors.
131        /// This field is only used when the destination shorthand field is set to 00B.
132        /// If the destination mode is set to physical, then bits 56 through 59 contain the APIC ID of the target processor for Pentium and P6 family processors and bits 56 through 63 contain the APIC ID of the target processor the for Pentium 4 and Intel Xeon processors.
133        /// If the destination mode is set to logical, the interpretation of the 8-bit destination field depends on the settings of the DFR and LDR registers of the local APICs in all the processors in the system
134        /// (see Section 11.6.2, “Determining IPI Destination”).
135        ///
136        /// 11.12.9 ICR Operation in x2APIC Mode
137        /// The destination ID field is expanded to 32 bits in x2APIC mode.
138        Destination OFFSET(24) NUMBITS(8) [],
139        /// Reserved
140        Reserved OFFSET(0) NUMBITS(24) []
141    ]
142}

144/// Interrupt Command Register (ICR) LOW using MMIO.
145/// - Address: FEE0 0300H (0 - 31)
146/// - Value after Reset: 0H
147pub type InterruptCommandRegisterLowMmio = ReadWrite<u32, INTERRUPT_COMMAND_LOW::Register>;
148
149/// A read-write copy of Interrupt Command Register (ICR) LOW.
150/// This behaves very similarly to a MMIO read-write register, but instead of doing a
151/// volatile read to MMIO to get the value for each function call, a copy of the
152/// register contents are stored locally in memory.
153pub type InterruptCommandRegisterLowLocal = LocalRegisterCopy<u32, INTERRUPT_COMMAND_LOW::Register>;
154
