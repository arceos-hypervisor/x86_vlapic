18        /// 111: Divide by 1

20            DivideBy2 = 0b0000,

22            DivideBy8 = 0b0010,
23            DivideBy16 = 0b0011,
24            DivideBy32 = 0b1000,
25            DivideBy64 = 0b1001,
26            DivideBy128 = 0b1010,

32/// Divide Configuration Register (FEE0 03E0H) using MMIO.

37/// A read-write copy of the Divide Configuration Register (FEE0 03E0H).
