1//! Figure 11-9. Error Status Register (ESR)
2//! 11.5.3 Error Handling
3//! The local APIC records errors detected during interrupt handling in the error status register (ESR).
4
5use tock_registers::LocalRegisterCopy;
6use tock_registers::fields::FieldValue;
7use tock_registers::register_bitfields;
8use tock_registers::registers::ReadWrite;
9
10register_bitfields! {
11    u32,
12    pub ERROR_STATUS [
13        /// Reserved
14        Reserved2 OFFSET(8) NUMBITS(24) [],
15        /// Bit 7: Illegal Register Address
16        /// Set when the local APIC is in xAPIC mode and software attempts to access a register that is reserved in the processor's local-APIC register-address space; see Table 10-1.
17        /// (The local-APIC register-address space comprises the 4 KBytes at the physical address specified in the IA32_APIC_BASE MSR.)
18        /// Used only on Intel Core, Intel Atom, Pentium 4, Intel Xeon, and P6 family processors.
19        IllegalRegisterAddress OFFSET(7) NUMBITS(1) [],
20        /// Bit 6: Receive Illegal Vector.
21        /// Set when the local APIC detects an illegal vector (one in the range 0 to 15) in an interrupt message it receives or in an interrupt generated locally from the local vector table or via a self IPI.
22        /// Such interrupts are not delivered to the processor; the local APIC will never set an IRR bit in the range 0 to 15.
23        ReceiveIllegalVector OFFSET(6) NUMBITS(1) [],
24        /// Bit 5: Send Illegal Vector.
25        /// Set when the local APIC detects an illegal vector (one in the range 0 to 15) in the message that it is sending.
26        /// This occurs as the result of a write to the ICR (in both xAPIC and x2APIC modes) or to SELF IPI register (x2APIC mode only) with an illegal vector.
27        /// If the local APIC does not support the sending of lowest-priority IPIs and software writes the ICR to send a lowest-priority IPI with an illegal vector, the local APIC sets only the “redirectable IPI” error bit.
28        ///  The interrupt is not processed and hence the “Send Illegal Vector” bit is not set in the ESR.
29        SendIllegalVector OFFSET(5) NUMBITS(1) [],
30        /// Bit 4: Redirectable IPI.
31        /// Set when the local APIC detects an attempt to send an IPI with the lowest-priority delivery mode and the local APIC does not support the sending of such IPIs.
32        /// This bit is used on some Intel Core and Intel Xeon processors. ]
33        /// As noted in Section 11.6.2, the ability of a processor to send a lowest-priority IPI is model-specific and should be avoided.
34        RedirectableIPI OFFSET(4) NUMBITS(1) [],
35        /// Bit 3: Receive Accept Error.
36        /// Set when the local APIC detects that the message it received was not accepted by any APIC on the APIC bus, including itself.
37        /// Used only on P6 family and Pentium processors.
38        ReceiveAcceptError OFFSET(3) NUMBITS(1) [],
39        /// Bit 2: Send Accept Error.
40        /// Set when the local APIC detects that a message it sent was not accepted by any APIC on the APIC bus.
41        /// Used only on P6 family and Pentium processors.
42        SendAcceptError OFFSET(2) NUMBITS(1) [],
43        /// Bit 1: Receive Checksum Error.
44        /// Set when the local APIC detects a checksum error for a message that it received on the APIC bus.
45        /// Used only on P6 family and Pentium processors.
46        ReceiveChecksumError OFFSET(1) NUMBITS(1) [],
47        /// Bit 0: Send Checksum Error.
48        /// Set when the local APIC detects a checksum error for a message that it sent on the APIC bus.
49        /// Used only on P6 family and Pentium processors.
50        SendChecksumError OFFSET(0) NUMBITS(1) [],
51    ]
52}

54/// Error Status Register (ESR) using MMIO.
55/// The local APIC records errors detected during interrupt handling in the error status register (ESR).
56/// - Address: FEE0 0280H
57/// - Value after reset: 0H
58pub type ErrorStatusRegisterMmio = ReadWrite<u32, ERROR_STATUS::Register>;
59
60/// A read-write copy of Error Status Register (ESR).
61/// This behaves very similarly to a MMIO read-write register, but instead of doing a
62/// volatile read to MMIO to get the value for each function call, a copy of the
63/// register contents are stored locally in memory.
64pub type ErrorStatusRegisterLocal = LocalRegisterCopy<u32, ERROR_STATUS::Register>;
65
66pub type ErrorStatusRegisterValue = FieldValue<u32, ERROR_STATUS::Register>;
